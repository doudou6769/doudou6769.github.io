<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>选号器</title>
</head>
<body>
    <div style="text-align: center;">
        <input type="text" value="0" style="width:100px;height:100px;font-size: 50px;text-align: center;" id="text">
        <input type="button" value="开始" id="start">
        <input type="button" value="停止" id="end">
    </div>
    <script>
        var timer;
        var start = document.getElementById("start");
        var end = document.getElementById("end");
        function startScroll(){
            var num = Math.floor(Math.random() * 60 + 1);
            document.getElementById("text").value = num;
            timer = setTimeout(startScroll,100);
        }
        
        function stopScroll(){
            clearTimeout(timer);
        }
        start.addEventListener("click",startScroll);
        end.addEventListener("click",stopScroll);
    </script>
</body>
